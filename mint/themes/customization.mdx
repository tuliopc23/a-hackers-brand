## Installing Custom Themes

```svelte
<!-- +layout.svelte -->
<script>
  import { ThemeProvider } from 'bubble-tea-party'
  import { themeStore } from 'bubble-tea-party/stores'
  import { myCustomTheme } from './themes/my-custom-theme.js'
  
  // Register custom theme
  themeStore.registerTheme(myCustomTheme)
  
  // Set as default
  themeStore.set(myCustomTheme)
</script>

<ThemeProvider theme={$themeStore}>
  <slot />
</ThemeProvider>
```

## Dynamic Theme Generation

Create themes programmatically:

```javascript
import { createTheme } from 'bubble-tea-party/theme'

function generateTheme(primaryColor, name) {
  return createTheme({
    name,
    colors: {
      primary: primaryColor,
      // Auto-generate complementary colors
      secondary: calculateComplement(primaryColor),
      accent: calculateTriadic(primaryColor)[0],
      // ... other colors derived from primary
    },
    effects: {
      glowIntensity: 'high',
      animationSpeed: 'normal'
    }
  })
}

// Usage
const hotPinkTheme = generateTheme('#ff1493', 'Hot Pink Cyber')
const electricBlueTheme = generateTheme('#0080ff', 'Electric Blue')
```


## Advanced Customization

### CSS Custom Properties

All theme values are available as CSS custom properties:

```css
.my-custom-component {
  background: var(--color-surface);
  color: var(--color-text);
  border: 1px solid var(--color-border);
  border-radius: var(--radius-md);
  padding: var(--spacing-md);
  font-family: var(--font-primary);
  transition: all var(--duration-normal) var(--easing-ease);
}

.my-custom-component:hover {
  box-shadow: var(--shadow-neon);
  transform: translateY(-2px);
}
```

### Theme Variants

Create multiple variants of the same base theme:

```javascript
import { createThemeVariant } from 'bubble-tea-party/theme'

const baseTheme = myCustomTheme

// Dark variant
const darkVariant = createThemeVariant(baseTheme, {
  name: 'Cyber Neon Dark',
  colors: {
    background: '#000000',
    surface: '#0a0a0a',
    text: '#ffffff'
  }
})

// Light variant (for accessibility)
const lightVariant = createThemeVariant(baseTheme, {
  name: 'Cyber Neon Light',
  colors: {
    background: '#ffffff',
    surface: '#f5f5f5',
    text: '#000000',
    primary: '#8a2be2', // Darker for contrast
    secondary: '#0066cc'
  }
})
```

## Theme Switching UI

Build a theme selector:

```svelte
<script>
  import { themeStore } from 'bubble-tea-party/stores'
  import { ThemeSelector, Button, Modal } from 'bubble-tea-party'
  
  let showThemeModal = false
  
  const availableThemes = [
    { id: 'cyberpunk-classic', name: 'Classic', preview: '#00ff41' },
    { id: 'neon-nights', name: 'Neon Nights', preview: '#ff0080' },
    { id: 'cyber-neon', name: 'Cyber Neon', preview: '#8a2be2' },
    // ... your custom themes
  ]
  
  function selectTheme(themeId) {
    themeStore.setById(themeId)
    showThemeModal = false
  }
</script>

<Button onClick={() => showThemeModal = true}>
  Customize Theme
</Button>

<Modal bind:open={showThemeModal} title="Choose Theme">
  <div class="theme-grid">
    {#each availableThemes as theme}
      <button 
        class="theme-card"
        style="--preview-color: {theme.preview}"
        on:click={() => selectTheme(theme.id)}
      >
        <div class="theme-preview"></div>
        <span>{theme.name}</span>
      </button>
    {/each}
  </div>
</Modal>

<style>
  .theme-grid {
    display: grid;
    grid-template-columns: repeat(auto-fit, minmax(120px, 1fr));
    gap: 1rem;
    padding: 1rem;
  }
  
  .theme-card {

  }
</style>
```

## Performance Optimization

Optimize theme switching performance:

```svelte
<script>
  import { themeStore } from 'bubble-tea-party/stores'
  import { writable } from 'svelte/store'
  
  // Debounce theme changes to avoid excessive re-renders
  const debouncedTheme = writable($themeStore)
  let themeTimeout
  
  themeStore.subscribe(theme => {
    clearTimeout(themeTimeout)
    themeTimeout = setTimeout(() => {
      debouncedTheme.set(theme)
    }, 100)
  })
</script>
```

## Testing Custom Themes

Validate your custom themes:

```javascript
import { validateTheme } from 'bubble-tea-party/theme'

const validation = validateTheme(myCustomTheme)

if (!validation.isValid) {
  console.error('Theme validation failed:', validation.errors)
  // Handle validation errors
}

// Check accessibility compliance
const accessibilityCheck = validateTheme(myCustomTheme, {
  checkContrast: true,
  checkColorBlindness: true
})
```
