## System Preference Detection

Automatically detect user's system preference:

```svelte
<script>
  import { onMount } from 'svelte'
  import { darkModeStore } from 'bubble-tea-party/stores'
  
  onMount(() => {
    // Check system preference
    const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)')
    
    // Set initial value
    darkModeStore.set(mediaQuery.matches)
    
    // Listen for changes
    const handleChange = (e) => {
      darkModeStore.set(e.matches)
    }
    
    mediaQuery.addEventListener('change', handleChange)
    
    return () => {
      mediaQuery.removeEventListener('change', handleChange)
    }
  })
</script>
```

## Dark Mode Persistence

Save user preference to localStorage:

```svelte
<script>
  import { onMount } from 'svelte'
  import { darkModeStore } from 'bubble-tea-party/stores'
  
  const STORAGE_KEY = 'cyberpunk-dark-mode'
  
  onMount(() => {
    // Load saved preference
    const saved = localStorage.getItem(STORAGE_KEY)
    if (saved !== null) {
      darkModeStore.set(JSON.parse(saved))
    }
    
    // Save changes
    return darkModeStore.subscribe(value => {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(value))
    })
  })
</script>
```

## Advanced Dark Mode Features

### Time-based Auto Switching

```svelte
<script>
  import { onMount } from 'svelte'
  import { darkModeStore } from 'bubble-tea-party/stores'
  
  let autoMode = false
  
  onMount(() => {
    if (autoMode) {
      const checkTime = () => {
        const hour = new Date().getHours()
        const shouldBeDark = hour < 6 || hour > 18
        darkModeStore.set(shouldBeDark)
      }
      
      checkTime()
      const interval = setInterval(checkTime, 60000) // Check every minute
      
      return () => clearInterval(interval)
    }
  })
</script>

<label>
  <input type="checkbox" bind:checked={autoMode} />
  Auto dark mode (6 PM - 6 AM)
</label>
```

### Gradual Transition

Smooth transition between modes:

```svelte
<script>
  import { tweened } from 'svelte/motion'
  import { darkModeStore } from 'bubble-tea-party/stores'
  
  const brightness = tweened(1, { duration: 1000 })
  
  $: brightness.set($darkModeStore ? 0.1 : 1)
</script>

<div 
  class="app"
  style="filter: brightness({$brightness})"
>
  <!-- App content -->
</div>
```
