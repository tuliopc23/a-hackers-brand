# Quick Start Guide

Get up and running with A Hacker's Brand in just 5 minutes. This guide will walk you through creating your first cyberpunk interface.

## Prerequisites

- **Node.js 18+** installed
- **Svelte 5.0+** project (SvelteKit recommended)
- Basic knowledge of Svelte/JavaScript

## Step 1: Installation

<CodeGroup>

```bash bun
bun add a-hacker-brand svelte @threlte/core @threlte/extras three lucide-svelte
```

```bash npm
npm install a-hacker-brand svelte @threlte/core @threlte/extras three lucide-svelte
```

```bash pnpm
pnpm add a-hacker-brand svelte @threlte/core @threlte/extras three lucide-svelte
```

</CodeGroup>

## Step 4: Build a Terminal Interface

Let's create a more complex example with multiple components:

**Terminal Status Bar Example**

<div className="component-mockup status-bar">Online</div>

```svelte
<StatusBar 
  variant="terminal"
  status="Online"
  encoding="UTF-8"
  fileType="Svelte"
  lineNumber={42}
  columnNumber={13}
/>
```

```svelte
<script>
  import { 
    Button, 
    Card, 
    Input, 
    StatusBar, 
    GlowText,
    Tabs,
    Tab
  } from 'a-hacker-brand';
  
  let command = '';
  let output = ['System initialized...', 'Ready for input.'];
  
  function executeCommand() {
    if (command.trim()) {
      output = [...output, `> ${command}`, `Executed: ${command}`];
      command = '';
    }
  }
</script>

<div class="terminal-app">
  <Card variant="terminal" class="terminal-window">
    <GlowText variant="terminal" color="matrix">
      HACKER TERMINAL v2.0
    </GlowText>
    
    <Tabs variant="terminal">
      <Tab title="Console">
        <div class="output">
          {#each output as line}
            <div class="output-line">{line}</div>
          {/each}
        </div>
        
        <div class="input-area">
          <Input 
            variant="terminal" 
            bind:value={command}
            placeholder="Enter command..."
            on:keydown={(e) => e.key === 'Enter' && executeCommand()}
          />
          <Button variant="terminal" on:click={executeCommand}>
            Execute
          </Button>
        </div>
      </Tab>
      
      <Tab title="System">
        <p>System monitoring interface...</p>
      </Tab>
    </Tabs>
  </Card>
  
  <StatusBar 
    variant="terminal"
    status="Online"
    encoding="UTF-8"
    fileType="Svelte"
    lineNumber={42}
    columnNumber={13}
  />
</div>

<style>
  .terminal-app {
    min-height: 100vh;
    background: #000;
    padding: 1rem;
    font-family: 'JetBrains Mono', monospace;
  }
  
  .terminal-window {
    max-width: 800px;
    margin: 0 auto;
  }
  
  .output {

    border-radius: 4px;

    overflow-y: auto;
    margin-bottom: 1rem;
    font-family: monospace;
  }

    color: #00ff41;

  }
</style>
```

## Step 5: Add Interactive Elements

Enhance your interface with interactive components:

```svelte
<script>
  import { 
    Button, 
    Card, 
    Switch, 
    Slider, 
    Progress,
    Alert,
    Toast,
    ColorPalette
  } from 'a-hacker-brand';
  
  let systemPower = true;
  let cpuUsage = 45;
  let showAlert = false;
  let showToast = false;
</script>

<div class="dashboard">
  <Card variant="glass">
    <h2>System Control Panel</h2>
    
    <!-- Power Control -->
    <div class="control-group">
      <label>System Power</label>
      <Switch 
        variant="glass" 
        bind:checked={systemPower}
        on:change={() => showToast = true}
      />
    </div>
    
    <!-- CPU Usage -->
    <div class="control-group">
      <label>CPU Usage: {cpuUsage}%</label>
      <Slider 
        variant="glass" 
        bind:value={cpuUsage}
        min={0}
        max={100}
      />
      <Progress variant="glass" value={cpuUsage} max={100} />
    </div>
    
    <!-- Actions -->
    <div class="actions">
      <Button 
        variant="glass" 
        on:click={() => showAlert = true}
      >
        Show Alert
      </Button>
      
      <Button 
        variant="terminal" 
        disabled={!systemPower}
      >
        Execute Task
      </Button>
    </div>
  </Card>
  
  <!-- Color Palette -->
  <Card variant="liquid">
    <h3>Theme Colors</h3>
    <ColorPalette 
      theme="both"
      layout="grid"
      interactive={true}
      on:colorSelect={(e) => console.log('Selected:', e.detail)}
    />
  </Card>
</div>

<!-- Alert -->
{#if showAlert}
  <Alert 
    variant="terminal" 
    type="warning"
    on:close={() => showAlert = false}
  >
    System alert: High CPU usage detected!
  </Alert>
{/if}

<!-- Toast -->
{#if showToast}
  <Toast 
    variant="glass"
    message="Power state changed"
    duration={3000}
    on:dismiss={() => showToast = false}
  />
{/if}

<style>
  .dashboard {
    display: grid;
    grid-template-columns: 2fr 1fr;
    gap: 2rem;
    padding: 2rem;
    min-height: 100vh;
    background: linear-gradient(135deg, #0a0a0a, #1a1a1a);
  }
  
  .control-group {
    margin-bottom: 1.5rem;
  }
  
  .control-group label {
    display: block;
    margin-bottom: 0.5rem;
    color: #00ff41;
    font-weight: 500;
  }
  
  .actions {
    display: flex;
    gap: 1rem;
    margin-top: 2rem;
  }
</style>
```

## Common Patterns

### Theme Switching

```svelte
<script>
  import { Button } from 'a-hacker-brand';
  
  let currentTheme = 'terminal';
  const themes = ['terminal', 'glass', 'liquid'];
  
  function switchTheme() {
    const currentIndex = themes.indexOf(currentTheme);
    currentTheme = themes[(currentIndex + 1) % themes.length];
  }
</script>

<Button variant={currentTheme} on:click={switchTheme}>
  Current: {currentTheme}
</Button>
```

### Responsive Layout

```svelte
<script>
  import { Card, Button } from 'a-hacker-brand';
</script>

<div class="responsive-grid">
  {#each Array(6) as _, i}
    <Card variant="glass">
      <h3>Card {i + 1}</h3>
      <Button variant="terminal">Action</Button>
    </Card>
  {/each}
</div>

<style>
  .responsive-grid {
    display: grid;
    grid-template-columns: repeat(auto-fit, minmax(280px, 1fr));
    gap: 1rem;
    padding: 1rem;
  }
  
  @media (max-width: 768px) {
    .responsive-grid {
      grid-template-columns: 1fr;
    }
  }
</style>
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Components not rendering">
    Ensure you have the correct peer dependencies installed:
    ```bash
    bun add svelte @threlte/core @threlte/extras three lucide-svelte
    ```
  </Accordion>
  
  <Accordion title="WebGL errors">
    WebGL components are optional. If you encounter errors:
    - Check browser WebGL support
    - Ensure hardware acceleration is enabled
    - Components will fallback to CSS-only versions
  </Accordion>
  
  <Accordion title="TypeScript errors">
    Make sure your `tsconfig.json` includes:
    ```json
    {
      "compilerOptions": {
        "moduleResolution": "bundler",
        "types": ["svelte"]
      }
    }
    ```
  </Accordion>
</AccordionGroup>
