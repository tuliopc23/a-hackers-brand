## Validation & States

### Built-in Validation

```svelte
<script>
  import { Input } from 'bubble-tea-party'
  
  let email = ''
  let isValid = false
  
  function validateEmail(value) {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)
  }
  
  $: isValid = validateEmail(email)
</script>

<Input 
  bind:value={email}
  label="Email Address"
  type="email"
  validation={{
    required: true,
    pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/,
    message: 'Please enter a valid email address'
  }}
  error={!isValid && email.length > 0}
  success={isValid && email.length > 0}
/>
```

## Props

| Property | Type | Default | Description |
|----------|------|---------|-------------|
| `variant` | `'default' \| 'cyberpunk' \| 'terminal' \| 'matrix'` | `'default'` | Input style variant |
| `type` | `string` | `'text'` | HTML input type |
| `label` | `string` | `''` | Input label text |
| `placeholder` | `string` | `''` | Placeholder text |
| `value` | `string` | `''` | Input value |
| `disabled` | `boolean` | `false` | Disable input |
| `readonly` | `boolean` | `false` | Make input readonly |
| `required` | `boolean` | `false` | Mark as required |
| `error` | `boolean` | `false` | Show error state |
| `success` | `boolean` | `false` | Show success state |
| `errorMessage` | `string` | `''` | Error message text |
| `successMessage` | `string` | `''` | Success message text |
| `leftIcon` | `string` | `''` | Left icon name |
| `rightIcon` | `string` | `''` | Right icon name |
| `maxLength` | `number` | `undefined` | Maximum character length |
| `showCounter` | `boolean` | `false` | Show character counter |
| `suggestions` | `string[]` | `[]` | Auto-complete suggestions |

## Form Integration

### With Form Validation

```svelte
<script>
  import { Input, Button, FormErrorBoundary } from 'bubble-tea-party'
  
  let formData = {
    username: '',
    email: '',
    password: ''
  }
  
  let errors = {}
  
  function validateForm() {
    errors = {}
    
    if (!formData.username) {
      errors.username = 'Username is required'
    }
    
    if (!formData.email.includes('@')) {
      errors.email = 'Valid email required'
    }
    
    if (formData.password.length < 8) {
      errors.password = 'Password must be 8+ characters'
    }
    
    return Object.keys(errors).length === 0
  }
  
  function handleSubmit() {
    if (validateForm()) {
      console.log('Access granted:', formData)
    }
  }
</script>

<FormErrorBoundary>
  <div class="cyber-form">
    <Input 
      bind:value={formData.username}
      label="Handle"
      variant="cyberpunk"
      error={!!errors.username}
      errorMessage={errors.username}
      required
    />
    
    <Input 
      bind:value={formData.email}
      label="Neural Link"
      type="email"
      variant="cyberpunk"
      error={!!errors.email}
      errorMessage={errors.email}
      required
    />
    
    <Input 
      bind:value={formData.password}
      label="Access Code"
      type="password"
      variant="cyberpunk"
      error={!!errors.password}
      errorMessage={errors.password}
      required
    />
    
    <Button 
      variant="neon" 
      onClick={handleSubmit}
      class="mt-4"
    >
      Jack In
    </Button>
  </div>
</FormErrorBoundary>
```
